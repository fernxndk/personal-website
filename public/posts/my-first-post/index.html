public int binarySearch(int[] sortedArray, targetIndex) {
	int low = 0;
	int high = sortedArray.length - 1;

	while (low <= high) {
		int mid = (low + high) / 2;
		int guess = sortedArray[mid];
		
		if (guess == targetIndex) {
			return mid;
		}
		if (guess > targetIndex) {
			high = mid - 1;
		}
		else {
			low = mid + 1;
		}
	}
	
	return None;
}
